/// Used with IXAudio2SourceVoice::SubmitSourceBuffer when submitting xWMA data
/// When streaming a file a few packets at a time, XAUDIO2_END_OF_STREAM
/// should be specified on the last packet.  Alternatively, the application may
/// call IXAudio2SourceVoice::Discontinuity after submitting the last packet.
/// Also, when streaming a file a few packets at a time, the application should
/// subtract pDecodedPacketCumulativeBytes[PacketCount-1] of the previous
/// buffer from all the entries of the currently submitted buffer.
#[derive(Clone, Copy, Debug)] #[repr(C, packed(1))] pub struct XAUDIO2_BUFFER_WMA {

    pDecodedPacketCumulativeBytes: *const u32,

    /// number WMA packets submitted, must be >= 1 and divide evenly into the respective XAUDIO2_BUFFER.AudioBytes
    PacketCount: u32,
}

macro_rules! interfaces {
    ($(
        $(#[doc = $interface_doc:literal])*
        $(#[iid = $iid:expr])?
        pub interface $interface:ident ( $interface_vtable:ident ) => unsafe $base:ident ( $base_vtable:ident ) {
            $(
                $(#[doc = $fn_doc:literal])*
                pub unsafe extern "system" fn $method:ident ( &self $(, $param_id:ident : $param_ty:ty )* $(,)? ) -> $return_ty:ty;
            )*
        }
    )*) => {$(
        $(#[doc = $interface_doc])*
        #[repr(C)] pub struct $interface ( * const $interface_vtable );
        #[repr(C)] pub struct $interface_vtable {
            pub base: $base_vtable,
            $(
                $(#[doc = $fn_doc])*
                pub $method : unsafe extern "system" fn(This: *const $interface $(, $param_id : $param_ty)*) -> $return_ty,
            )*
        }
        impl $interface {
            $(
                $(#[doc = $fn_doc])*
                pub unsafe fn $method(&self $(, $param_id : $param_ty)*) -> $return_ty {
                    unsafe { ((*self.0).$method)(self $(, $param_id)*) }
                }
            )*
        }
        impl core::ops::Deref for $interface {
            type Target = $base;
            fn deref(&self) -> &Self::Target { unsafe { core::mem::transmute(self) } }
        }
        interfaces!(@iid $interface $(#[iid = $iid])?);
    )*};

    (@iid $interface:ident                   ) => { unsafe impl mcom::AsIUnknown for $interface { fn as_iunknown(&self) -> &IUnknown { self } } };
    (@iid $interface:ident #[iid = $iid:expr]) => { impl winapi::Interface for $interface { fn uuidof() -> winapi::shared::guiddef::GUID { $iid.into() } } };
}

    /// in the XAudio2 engine.  This interface should be implemented by
    /// the client.  XAudio2 will call these methods via the interface
    /// pointer provided by the client when it calls XAudio2Create() or
    /// IXAudio2::Initialize().
    pub interface IXAudio2EngineCallback(IXAudio2EngineCallbackVtbl) => unsafe IUnknown(IUnknownVtbl) {
        /// Called by XAudio2 just before an audio processing pass begins.
        pub unsafe extern "system" fn OnProcessingPassStart(&self) -> ();

        /// Called just after an audio processing pass ends.
        pub unsafe extern "system" fn OnProcessingPassEnd(&self) -> ();

        /// Called in the event of a critical system error which requires XAudio2
        /// to be closed down and restarted.  The error code is given in Error.
        pub unsafe extern "system" fn OnCriticalError(&self, error: HResult) -> ();
    }


    /// Client notification interface for voice events.
    ///
    /// Contains methods to notify the client when certain events happen
    /// in an XAudio2 voice.  This interface should be implemented by the
    /// client.  XAudio2 will call these methods via an interface pointer
    /// provided by the client in the IXAudio2::CreateSourceVoice() call.
    pub interface IXAudio2VoiceCallback(IXAudio2VoiceCallbackVtbl) => unsafe IUnknown(IUnknownVtbl) {
        /// Called just before this voice's processing pass begins.
        pub unsafe extern "system" fn OnVoiceProcessingPassStart(&self) -> ();

        /// Called just after this voice's processing pass ends.
        pub unsafe extern "system" fn OnVoiceProcessingPassEnd(&self) -> ();

        /// Called when this voice has just finished playing a buffer stream
        /// (as marked with the XAUDIO2_END_OF_STREAM flag on the last buffer).
        pub unsafe extern "system" fn OnStreamEnd(&self) -> ();

        /// Called when this voice is about to start processing a new buffer.
        pub unsafe extern "system" fn OnBufferStart(&self, pBufferContext: *mut c_void) -> ();

        /// Called when this voice has just finished processing a buffer.
        /// The buffer can now be reused or destroyed.
        pub unsafe extern "system" fn OnBufferEnd(&self, pBufferContext: *mut c_void) -> ();

        /// Called when this voice has just reached the end position of a loop.
        pub unsafe extern "system" fn OnLoopEnd(&self, pBufferContext: *mut c_void) -> ();

        /// Called in the event of a critical error during voice processing,
        /// such as a failing XAPO or an error from the hardware XMA decoder.
        /// The voice may have to be destroyed and re-created to recover from
        /// the error.  The callback arguments report which buffer was being
        /// processed when the error occurred, and its HResult code.
        pub unsafe extern "system" fn OnVoiceError(&self, pBufferContext: *mut c_void, error: HResult) -> ();
    }
}

/// Calculate the argument to SetVolume() from a decibel value
#[inline] pub fn XAudio2DecibelsToAmplitudeRatio(Decibels: f64) -> f32 {
    f64::powf(10.0, Decibels / 20.0) as f32
}

/// Recover a volume in decibels from an amplitude factor
#[inline] pub fn XAudio2AmplitudeRatioToDecibels(Volume: f64) -> f32 {
    (20.0 * f64::log10(Volume)) as f32
}

/// Calculate the argument to SetFrequencyRatio() from a semitone value
#[inline] pub fn XAudio2SemitonesToFrequencyRatio(Semitones: f64) -> f32 {
    // FrequencyRatio = 2 ^ Octaves
    //                = 2 ^ (Semitones / 12)
    f64::powf(2.0, Semitones / 12.0) as f32
}

/// Recover a pitch in semitones from a frequency ratio
#[inline] pub fn XAudio2FrequencyRatioToSemitones(FrequencyRatio: f64) -> f32 {
    // Semitones = 12 * log2(FrequencyRatio)
    //           = 12 * log2(10) * log10(FrequencyRatio)
    (39.86313713864835 * f64::log10(FrequencyRatio)) as f32
}

/// Convert from filter cutoff frequencies expressed in Hertz to the radian
/// frequency values used in XAUDIO2_FILTER_PARAMETERS.Frequency
#[inline] pub fn XAudio2CutoffFrequencyToRadians(CutoffFrequency: f64, SampleRate: u32) -> f32 {
    let SampleRate = f64::from(SampleRate);
    if CutoffFrequency < 0.0 || 6.0 * CutoffFrequency > SampleRate {
        0.0 // Filter is only valid up to SampleRate/6
    } else {
        (2.0 * f64::sin(core::f64::consts::PI * CutoffFrequency / SampleRate)) as f32
    }
}

/// Convert from radian frequencies back to absolute frequencies in Hertz
#[inline] pub fn XAudio2RadiansToCutoffFrequency(Radians: f64, SampleRate: f64) -> f32 {
    (SampleRate * f64::asin(Radians / 2.0) / core::f64::consts::PI) as f32
}

/// Top-level function that creates an XAudio2 instance.
///
/// On Windows this is just an inline function that calls CoCreateInstance
/// and Initialize.  The arguments are described above, under Initialize,
/// except that the XAUDIO2_DEBUG_ENGINE flag can be used here to select
/// the debug version of XAudio2.
///
/// On Xbox, this function is implemented in the XAudio2 library, and the
/// XAUDIO2_DEBUG_ENGINE flag has no effect; the client must explicitly
/// link with the debug version of the library to obtain debug behavior.
///
/// ### ⚠️ Safety ⚠️
/// *   `ppXAudio2` must be a valid non-null COM ptr
/// *   `Flags` may need to be valid
/// *   Per [Known Issues: XAudio 2.7](https://web.archive.org/web/20181227221804/https://blogs.msdn.microsoft.com/chuckw/2015/10/09/known-issues-xaudio-2-7/), the XAudio2 DLL may be unloaded before all XAudio2 objects are destroyed
#[cfg(not(xbox))] #[inline] pub unsafe fn XAudio2Create(
    ppXAudio2:          *mut *mut IXAudio2,
    Flags:              u32,
    XAudio2Processor:   XAUDIO2_PROCESSOR,
) -> HResult {
    let xaudio2 = unsafe { mcom::Rc::<IXAudio2>::co_create(
        if Flags & XAUDIO2_DEBUG_ENGINE != 0 { CLSID_XAudio2_Debug } else { CLSID_XAudio2 }.into(),
        None,
    )};

    match xaudio2 {
        Ok(xaudio2) => {
            let hr = unsafe { xaudio2.Initialize(Flags, XAudio2Processor) };
            if !hr.is_error() { unsafe { *ppXAudio2 = xaudio2.into_raw() } }
            hr
        },
        Err(hr) => hr.hresult().into(),
    }
}

/// Object type values used by XAudio2 for its internal memory allocations.
/// On Xbox these become calls to XMemAlloc.  If the client provides its

    Generic,      // Misc allocations that don't fall into any of the categories below
    Engine,       // General XAudio2 engine data
    Voice,        // General per-voice data
    BufferPool,   // Management of internal buffers
    EffectChain,  // Management of effect chains
    MixMatrix,    // Management of send level matrices
    LEAP,         // XAudio2's internal audio processor
    Pipeline,     // DSP routines built into XAudio2
    //Count         // Count of all object types
}
