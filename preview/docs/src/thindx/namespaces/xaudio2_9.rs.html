interfaces! {
    /// This interface extends IXAudio2 with additional functionality.
    /// Use IXAudio2::QueryInterface to obtain a pointer to this interface.
    pub(crate) interface IXAudio2Extension(IXAudio2ExtensionVtbl) => unsafe IUnknown(IUnknownVtbl) {
        /// Returns the processing quantum
        /// quantumMilliseconds = (1000.0f * quantumNumerator / quantumDenominator)
        ///
        /// ### Arguments
        /// * `quantumNumerator`    - Quantum numerator
        /// * `quantumDenominator`  - Quantum denominator
        pub unsafe extern "system" fn GetProcessingQuantum(&self, quantumNumerator: *mut u32, quantumDenominator: *mut u32) -> ();

        /// Returns the number of the processor used by XAudio2
        ///
        /// ### Arguments
        /// * `processor`           - Non-zero Processor number
        pub unsafe extern "system" fn GetProcessor(&self, processor: *mut PROCESSOR) -> ();
    }
    //#cpp2rust IXAudio2Extension       = xaudio2_9::IXAudio2Extension
    //#cpp2rust IXAudio2ExtensionVtbl   = xaudio2_9::IXAudio2ExtensionVtbl
}
